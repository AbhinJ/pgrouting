61  std::deque<Path> Yen(
62  G &graph,
63  int64_t start_vertex,
64  int64_t end_vertex,
65  size_t K,
66  bool heap_paths) {
67  /*
68  * No path: already in destination
69  */
70  if ((start_vertex == end_vertex) || (K == 0)) {
71  return std::deque<Path>();
72  }
73 
74  /*
75  * no path: disconnected vertices
76  */
77  if (!graph.has_vertex(start_vertex)
78  || !graph.has_vertex(end_vertex)) {
79  return std::deque<Path>();
80  }
81 
82  /*
83  * clean up containers
84  */
85  clear();
86 
87 
88  v_source = graph.get_V(start_vertex);
89  v_target = graph.get_V(end_vertex);
90  m_start = start_vertex;
91  m_end = end_vertex;
92  m_K = K;
93  m_heap_paths = heap_paths;
94 
95 
96  executeYen(graph);
97 
98  auto paths = get_results();
99 
100  if (!m_heap_paths && paths.size() > m_K) paths.resize(m_K);
101 
102  return paths;
103  }
104

106  m_Heap.clear();
107  m_ResultSet.clear();

112  class Visitor {
113  public:
114  virtual ~Visitor() {}

116  virtual void on_insert_first_solution(const Path) const {

125  protected:
127  void executeYen(G &graph) {
128  clear();
129  curr_result_path = getFirstSolution(graph);
130  m_vis->on_insert_first_solution(curr_result_path);
131 
132  if (m_ResultSet.size() == 0) return; // no path found
133 
134  while (m_ResultSet.size() < m_K) {
135  doNextCycle(graph);
136  if (m_Heap.empty()) break;
137  curr_result_path = *m_Heap.begin();
138  m_ResultSet.insert(curr_result_path);
139  m_Heap.erase(m_Heap.begin());
140  }
141  }
142 
144 
147  protected:
148  Path getFirstSolution(G &graph) {
149  Path path;
150 
151  Pgr_dijkstra< G > fn_dijkstra;
152  path = fn_dijkstra.dijkstra(graph, m_start, m_end);
153 
154  if (path.empty()) return path;
155  m_ResultSet.insert(path);
156  return path;
157  }
158 
159 
160  protected:
162  void doNextCycle(G &graph) {
163  for (unsigned int i = 0; i < curr_result_path.size(); ++i) {
164  int64_t spurNodeId = curr_result_path[i].node;
165 
166  auto rootPath = curr_result_path.getSubpath(i);
167 
168  for (const auto &path : m_ResultSet) {
169  if (path.isEqual(rootPath)) {
170  if (path.size() > i + 1) {
171  graph.disconnect_edge(path[i].node, // from
172  path[i + 1].node); // to
173  }
174  }
175  }
176 
177  removeVertices(graph, rootPath);
178 
179  Pgr_dijkstra< G > fn_dijkstra;
180  auto spurPath = fn_dijkstra.dijkstra(graph, spurNodeId, m_end);
181 
182  if (spurPath.size() > 0) {
183  rootPath.appendPath(spurPath);
184  m_Heap.insert(rootPath);
185  m_vis->on_insert_to_heap(rootPath);
186  }
187 
188  graph.restore_graph();
189  }
190  }
191 
192  protected:
194  void removeVertices(G &graph, const Path &subpath) {
195  for (const auto &e : subpath)
196  graph.disconnect_vertex(e.node);
197  }
198 
199  std::deque<Path> get_results() {
200  if (this->m_ResultSet.empty()) {
201  return std::deque<Path>();
202  }
203 
204  std::deque<Path> paths(m_ResultSet.begin(), m_ResultSet.end());
205 
206  if (m_heap_paths && !m_Heap.empty()) {
207  paths.insert(paths.end(), m_Heap.begin(), m_Heap.end());
208  }
209  pgassert(!paths.empty());
210 
211  std::sort(paths.begin(), paths.end(), compPathsLess());
212 
213  return paths;
214  }
