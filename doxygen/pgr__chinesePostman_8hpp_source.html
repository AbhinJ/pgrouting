108 PgrDirectedChPPGraph::PgrDirectedChPPGraph(
109  const pgr_edge_t *dataEdges,
110  const size_t totalEdges) :
111  totalDeg(0), totalCost(0), vertices(),
112  edgeToIdx(), originalEdges(),
113  resultGraph(), VToVecid(), edgeVisited(),
114  pathStack(), resultPath(),
115  edges(), sources(), targets()
116 {
117  pgassert(totalEdges > 0);
118  pgassert(pathStack.empty());
119 
120  pgassert(originalEdges.empty());
121  startPoint = dataEdges[0].source;
122  for (size_t i = 0; i < totalEdges; ++i) {
123  if (dataEdges[i].cost > 0) {
124  auto edge(dataEdges[i]);
125  edge.reverse_cost = -1.0;
126  totalCost += edge.cost;
127  originalEdges.push_back(edge);
128  vertices += dataEdges[i].source;
129  vertices += dataEdges[i].target;
130  }
131  if (dataEdges[i].reverse_cost > 0) {
132  auto edge(dataEdges[i]);
133  std::swap(edge.source, edge.target);
134  std::swap(edge.cost, edge.reverse_cost);
135  edge.reverse_cost = -1.0;
136  totalCost += edge.cost;
137  originalEdges.push_back(edge);
138  vertices += dataEdges[i].source;
139  vertices += dataEdges[i].target;
140  pgassert(dataEdges[i].source == edge.target);
141  pgassert(dataEdges[i].target == edge.source);
142  }
143  }
144 
145  // calcu deg & build part of edges
146  std::map<int64_t, int> deg;
147  size_t i(0);
148  for (const auto &e : originalEdges) {
149  pgassert(e.cost > 0);
150  /* has out going edge */
151  deg[e.source]++;
152  /* has out incoming edge */
153  deg[e.target]--;
154 
155  auto current_edge(std::make_pair(e.source, e.target));
156  if (edgeToIdx.find(current_edge) == edgeToIdx.end()) {
157  edgeToIdx[current_edge] = &e;
158  } else {
159  if (edgeToIdx[current_edge]->cost > e.cost) {
160  edgeToIdx[current_edge] = &e;
161  }
162  }
163 
164  pgr_costFlow_t edge;
165  edge.edge_id = e.id;
166  edge.reverse_capacity = -1;
167  edge.reverse_cost = -1.0;
168  edge.source = e.source;
169  edge.target = e.target;
170  edge.capacity = (std::numeric_limits<int32_t>::max)();
171  edge.cost = e.cost;
172  edges.push_back(edge);
173  ++i;
174  }
175 
176  superSource = deg.rbegin()->first + 1;
177  superTarget = deg.rbegin()->first + 2;
178 
179  sources.insert(superSource);
180  targets.insert(superTarget);
181 
182  // build full edges
183  std::map<int64_t, int>::iterator iter;
184  totalDeg = 0;
185  for (iter = deg.begin(); iter != deg.end(); ++iter) {
186  int64_t p = iter->first;
187  int d = iter->second;
188  if (d == 0)
189  continue;
190  if (d > 0)
191  totalDeg += d;
192  pgr_costFlow_t edge;
193  edge.reverse_capacity = -1;
194  edge.reverse_cost = -1.0;
195  edge.cost = 0.0;
196  if (d > 0)
197  edge.capacity = d;
198  else
199  edge.capacity = -d;
200  edge.edge_id = 0;
201  if (d > 0)
202  edge.source = p, edge.target = superTarget;
203  if (d < 0)
204  edge.source = superSource, edge.target = p;
205  edges.push_back(edge);
206  }
207 
208  PgrCostFlowGraph flowGraph(edges, sources, targets);
209  pgassert(pathStack.empty());
210  try {
211  pgrouting::graph::PgrCostFlowGraph digraph1(edges, sources, targets);
212  auto minAddedCost = digraph1.MinCostMaxFlow();
213  auto maxFlow = digraph1.GetMaxFlow();
214  m_cost = (maxFlow == totalDeg)? minAddedCost + totalCost : -1.0;
215  } catch (...) {
216  m_cost = -1;
217  }
218  setPathEdges(flowGraph);
219 }

222 void
223 PgrDirectedChPPGraph::setPathEdges(graph::PgrCostFlowGraph &flowGraph) {
224  pgassert(pathStack.empty());
225  resultPath.clear();
226  if (m_cost == -1) return;
227  // catch new edges
228  try {
229  flowGraph.MinCostMaxFlow();
230  flowGraph.GetMaxFlow();
231  std::vector<pgr_flow_t> addedEdges = flowGraph.GetFlowEdges();
232  resultEdges = originalEdges;
233  for (auto &flow_t : addedEdges) {
234  if (flow_t.source != superSource && flow_t.source != superTarget
235  && flow_t.target != superSource && flow_t.target != superTarget) {
236  auto current_edge(std::make_pair(flow_t.source, flow_t.target));
237  pgr_edge_t newEdge = *edgeToIdx[current_edge];
238  /* adding edges that need to be traversed twice */
239  while (flow_t.flow--) resultEdges.push_back(newEdge);
240  }
241  }
242  } catch (...) {
243  resultPath.clear();
244  return;
245  }
246 
247  pgassert(pathStack.empty());
248  BuildResultGraph();
249 
250  pgassert(pathStack.empty());
251  pgassert(edgeVisited.empty());
252 
253 
254  EulerCircuitDFS(startPoint);
255 
256  if (!(vertices - vertexVisited).empty()) {
257  resultPath.clear();
258  return;
259  }
260  pgassert(!pathStack.empty());
261 
262  BuildResultPath();
263 
264 
265  return;
266 }

268 void
269 PgrDirectedChPPGraph::BuildResultPath() {
270  if (pathStack.empty()) return;
271  pgassert(resultPath.empty());
272 
273  int64_t preNode = pathStack.top();
274  pathStack.pop();
275 
276  General_path_element_t newElement;
277  while (!pathStack.empty()) {
278  int64_t nowNode = pathStack.top();
279  pathStack.pop();
280 
281  auto edge_t = *edgeToIdx[std::make_pair(preNode, nowNode)];
282  newElement.node = edge_t.source;
283  newElement.edge = edge_t.id;
284  newElement.cost = edge_t.cost;
285  if (resultPath.empty()) {
286  /* adding the first row because is a cycle */
287  newElement.seq = 1;
288  newElement.agg_cost = 0.0;
289  } else {
290  newElement.seq = resultPath.back().seq + 1;
291  newElement.agg_cost = resultPath.back().agg_cost + resultPath.back().cost;
292  }
293  resultPath.push_back(newElement);
294  preNode = nowNode;
295  }
296  newElement.node = preNode;
297  newElement.edge = -1;
298  newElement.cost = 0;
299  if (resultPath.empty()) {
300  newElement.seq = 1;
301  newElement.agg_cost = 0.0;
302  } else {
303  newElement.seq = resultPath.back().seq + 1;
304  newElement.agg_cost =
305  resultPath.back().agg_cost + resultPath.back().cost;
306  }
307  resultPath.push_back(newElement);
308 }

313 bool
314 PgrDirectedChPPGraph::EulerCircuitDFS(int64_t vertex) {
315  for (const auto e : resultGraph[VToVecid[vertex]].second) {
316  if (!edgeVisited.has(e)) {
317  edgeVisited += e;
318  EulerCircuitDFS(resultEdges[e].target);
319  }
320  }
321  pathStack.push(vertex);
322  vertexVisited += vertex;
323  return true;
324 }

326 void
327 PgrDirectedChPPGraph::BuildResultGraph() {
328  pgassert(resultGraph.empty());
329  pgassert(VToVecid.empty());
330  resultGraph.clear();
331  VToVecid.clear();
332  size_t e_id(0);
333  for (const auto &edge_t : resultEdges) {
334  if (VToVecid.find(edge_t.source) == VToVecid.end()) {
335  VToVecid[edge_t.source] = resultGraph.size();
336  resultGraph.resize(resultGraph.size() + 1);
337  }
338  size_t vid = VToVecid[edge_t.source];
339  resultGraph[vid].second.push_back(e_id);
340  resultGraph[vid].first = edge_t.source;
341  ++e_id;
342  }
343 }
