88 PgrDirectedChPPGraph::PgrDirectedChPPGraph(
89  const pgr_edge_t *dataEdges,
90  const size_t totalEdges) :
91  totalCost(0) {
92  resultEdges.clear();
93  for (size_t i = 0; i < totalEdges; i++) {
94  pgr_edge_t edge;
95  edge.id = dataEdges[i].id;
96  edge.source = dataEdges[i].source;
97  edge.target = dataEdges[i].target;
98  edge.reverse_cost = -1.0;
99  if (dataEdges[i].cost > 0) {
100  startPoint = edge.source;
101  edge.cost = dataEdges[i].cost;
102  totalCost += edge.cost;
103  resultEdges.push_back(edge);
104  }
105  if (dataEdges[i].reverse_cost > 0) {
106  std::swap(edge.source, edge.target);
107  edge.cost = dataEdges[i].reverse_cost;
108  totalCost += edge.cost;
109  resultEdges.push_back(edge);
110  }
111  }
112 
113  std::vector<pgr_costFlow_t> edges;
114  std::set<int64_t> sources;
115  std::set<int64_t> targets;
116 
117  // calcu deg & build part of edges
118  std::map<int64_t, int> deg;
119  for (size_t i = 0; i < resultEdges.size(); i++) {
120  deg[resultEdges[i].source]++;
121  deg[resultEdges[i].target]--;
122 
123  if (edgeToId.find(std::make_pair(resultEdges[i].source, resultEdges[i].target)) ==
124  edgeToId.end()) {
125  edgeToId.insert(std::make_pair(std::make_pair(resultEdges[i].source, resultEdges[i].target), i));
126  } else {
127  if (resultEdges[edgeToId[
128  std::make_pair(resultEdges[i].source, resultEdges[i].target)]].cost >
129  resultEdges[i].cost) {
130  edgeToId[std::make_pair(resultEdges[i].source, resultEdges[i].target)] = i;
131  }
132  }
133 
134  pgr_costFlow_t edge;
135  edge.edge_id = resultEdges[i].id;
136  edge.reverse_capacity = -1;
137  edge.reverse_cost = -1.0;
138  edge.source = resultEdges[i].source;
139  edge.target = resultEdges[i].target;
140  edge.capacity = (std::numeric_limits<int32_t>::max)();
141  edge.cost = resultEdges[i].cost;
142  edges.push_back(edge);
143  }
144 
145  // find superSource & superTarget
146  superSource = superTarget = -1;
147  int64_t iPointId = 1;
148  while (superSource == -1 || superTarget == -1) {
149  if (deg.find(iPointId) == deg.end()) {
150  if (superSource == -1)
151  superSource = iPointId;
152  else
153  superTarget = iPointId;
154  }
155  iPointId++;
156  }
157  sources.insert(superSource);
158  targets.insert(superTarget);
159 
160  // build full edges
161  std::map<int64_t, int>::iterator iter;
162  totalDeg = 0;
163  for (iter = deg.begin(); iter != deg.end(); ++iter) {
164  int64_t p = iter->first;
165  int d = iter->second;
166  if (d == 0)
167  continue;
168  if (d > 0)
169  totalDeg += d;
170  pgr_costFlow_t edge;
171  edge.reverse_capacity = -1;
172  edge.reverse_cost = -1.0;
173  edge.cost = 0.0;
174  if (d > 0)
175  edge.capacity = d;
176  else
177  edge.capacity = -d;
178  edge.edge_id = 0;
179  if (d > 0)
180  edge.source = p, edge.target = superTarget;
181  if (d < 0)
182  edge.source = superSource, edge.target = p;
183  edges.push_back(edge);
184  }
185 
186  PgrCostFlowGraph graph(edges, sources, targets);
187  flowGraph = graph;
188 }

198 std::vector<General_path_element_t>
199 PgrDirectedChPPGraph::GetPathEdges() {
200  // catch new edges
201  std::vector<pgr_flow_t> addedEdges = flowGraph.GetFlowEdges();
202  for (auto &flow_t : addedEdges) {
203  if (flow_t.source != superSource && flow_t.source != superTarget) {
204  if (flow_t.target != superSource && flow_t.target != superTarget) {
205  pgr_edge_t newEdge;
206  newEdge.source = flow_t.source;
207  newEdge.target = flow_t.target;
208  newEdge.id =
209  resultEdges[edgeToId[std::make_pair(
210  newEdge.source,
211  newEdge.target)]].id;
212  newEdge.cost =
213  resultEdges[edgeToId[std::make_pair(
214  newEdge.source,
215  newEdge.target)]].cost;
216  newEdge.reverse_cost = -1.0;
217  while (flow_t.flow--)
218  resultEdges.push_back(newEdge);
219  }
220  }
221  }
222 
223  BuildResultGraph();
224 
225  EulerCircuitDFS(startPoint);
226 
227  BuildResultPath();
228 
229  if (!JudgeCoveredAllEdges())
230  resultPath.clear();
231  return resultPath;
232 }

234 void
235 PgrDirectedChPPGraph::BuildResultPath() {
236  if (pathStack.empty())
237  return;
238  int64_t preNode = pathStack.top();
239  pathStack.pop();
240 
241  General_path_element_t newElement;
242  while (!pathStack.empty()) {
243  int64_t nowNode = pathStack.top();
244  pathStack.pop();
245  pgr_edge_t edge_t =
246  resultEdges[edgeToId[std::make_pair(preNode, nowNode)]];
247  newElement.node = edge_t.source;
248  newElement.edge = edge_t.id;
249  newElement.cost = edge_t.cost;
250  if (resultPath.empty()) {
251  newElement.seq = 1;
252  newElement.agg_cost = 0.0;
253  } else {
254  newElement.seq = resultPath.back().seq + 1;
255  newElement.agg_cost =
256  resultPath.back().agg_cost + resultPath.back().cost;
257  }
258  resultPath.push_back(newElement);
259  preNode = nowNode;
260  }
261  newElement.node = preNode;
262  newElement.edge = -1;
263  newElement.cost = 0;
264  if (resultPath.empty()) {
265  newElement.seq = 1;
266  newElement.agg_cost = 0.0;
267  } else {
268  newElement.seq = resultPath.back().seq + 1;
269  newElement.agg_cost =
270  resultPath.back().agg_cost + resultPath.back().cost;
271  }
272  resultPath.push_back(newElement);
273 }

279 PgrDirectedChPPGraph::EulerCircuitDFS(int64_t p) {
280  for (auto iter = resultGraph[VToVecid[p]].second.begin();
281  iter != resultGraph[VToVecid[p]].second.end();
282  ++iter) {
283  if (!edgeVisited[*iter]) {
284  edgeVisited[*iter] = true;
285  EulerCircuitDFS(resultEdges[*iter].target);
286  }
287  }
288  pathStack.push(p);

292 void
293 PgrDirectedChPPGraph::BuildResultGraph() {
294  resultGraph.clear();
295  VToVecid.clear();
296  edgeVisited.clear();
297  for (size_t i = 0; i < resultEdges.size(); i++) {
298  pgr_edge_t edge_t = resultEdges[i];
299  edgeVisited.push_back(false);
300  if (VToVecid.find(edge_t.source) == VToVecid.end()) {
301  VToVecid.insert(
302  std::pair<int64_t, size_t>(
303  edge_t.source,
304  resultGraph.size()));
305  resultGraph.resize(resultGraph.size() + 1);
306  }
307  size_t vid = VToVecid[edge_t.source];
308  resultGraph[vid].second.push_back(i);
309  resultGraph[vid].first = edge_t.source;
310  }
311 }
