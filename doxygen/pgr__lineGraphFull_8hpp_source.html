228  void apply_transformation(
229  const pgrouting::DirectedGraph& digraph) {
230  V_i vertexIt, vertexEnd;
231  EO_i e_outIt, e_outEnd;
232  EI_i e_inIt, e_inEnd;
233 
234  // For every vertex in the original graph, create a line graph
235  // using the edges connected to that vertex
236  for (boost::tie(vertexIt, vertexEnd) = boost::vertices(digraph.graph);
237  vertexIt != vertexEnd; vertexIt++) {
238  V vertex = *vertexIt;
239  auto vertex_id = digraph[vertex].id;
240  for (boost::tie(e_outIt, e_outEnd) =
241  boost::out_edges(vertex, digraph.graph);
242  e_outIt != e_outEnd; e_outIt++) {
243  auto out_edge_id = digraph.graph[*e_outIt].id;
244  insert_vertex(vertex_id, out_edge_id);
245  }
246 
247  for (boost::tie(e_inIt, e_inEnd) =
248  boost::in_edges(vertex, digraph.graph);
249  e_inIt != e_inEnd; e_inIt++) {
250  auto in_edge_id = digraph.graph[*e_inIt].id;
251  insert_vertex(vertex_id, in_edge_id);
252 
253  for (boost::tie(e_outIt, e_outEnd) =
254  boost::out_edges(vertex, digraph.graph);
255  e_outIt != e_outEnd; e_outIt++) {
256  auto out_edge_id = digraph.graph[*e_outIt].id;
257 
258  ++m_num_edges;
259 
260  graph_add_edge(
261  m_num_edges,
262  vertex_id,
263  vertex_id,
264  in_edge_id,
265  out_edge_id);
266  }
267  }
268  }
269 
270  // Connect all of the line graphs that were just created using the
271  // edges from the original graph
272  for (boost::tie(vertexIt, vertexEnd) =
273  boost::vertices(digraph.graph);
274  vertexIt != vertexEnd; vertexIt++) {
275  V vertex = *vertexIt;
276  auto vertex_id = digraph[vertex].id;

278  for (boost::tie(e_inIt, e_inEnd) =
279  boost::in_edges(vertex, digraph.graph);
