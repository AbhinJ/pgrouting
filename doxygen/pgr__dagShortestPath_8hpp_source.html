59  Path dag(
60  G &graph,
61  int64_t start_vertex,
62  int64_t end_vertex,
63  bool only_cost = false) {
64  clear();
65 
66  // adjust predecessors and distances vectors
67  predecessors.resize(graph.num_vertices());
68  distances.resize(
69  graph.num_vertices(),
70  std::numeric_limits<double>::infinity());
71 
72 
73  if (!graph.has_vertex(start_vertex)
74  || !graph.has_vertex(end_vertex)) {
75  return Path(start_vertex, end_vertex);
76  }
77 
78  // get the graphs source and target
79  auto v_source(graph.get_V(start_vertex));
80  auto v_target(graph.get_V(end_vertex));
81 
82  // perform the algorithm
83  dag_1_to_1(graph, v_source, v_target);
84 
85  // get the results
86  return Path(
87  graph,
88  v_source, v_target,
89  predecessors, distances,
90  only_cost, true);
91  }

95  std::deque<Path> dag(
96  G &graph,
97  int64_t start_vertex,
98  const std::vector< int64_t > &end_vertex,
99  bool only_cost) {
100  // adjust predecessors and distances vectors
101  clear();
102  size_t n_goals = (std::numeric_limits<size_t>::max)();
103  predecessors.resize(graph.num_vertices());
104  distances.resize(
105  graph.num_vertices(),
106  std::numeric_limits<double>::infinity());
107 
108 
109  // get the graphs source and target
110  if (!graph.has_vertex(start_vertex))
111  return std::deque<Path>();
112  auto v_source(graph.get_V(start_vertex));
113 
114  std::set< V > s_v_targets;
115  for (const auto &vertex : end_vertex) {
116  if (graph.has_vertex(vertex)) {
117  s_v_targets.insert(graph.get_V(vertex));
118  }
119  }
120 
121  std::vector< V > v_targets(s_v_targets.begin(), s_v_targets.end());
122  // perform the algorithm
123  dag_1_to_many(graph, v_source, v_targets, n_goals);
124 
125  std::deque< Path > paths;
126  // get the results // route id are the targets
127  paths = get_paths(graph, v_source, v_targets, only_cost);
128 
129  std::stable_sort(paths.begin(), paths.end(),
130  [](const Path &e1, const Path &e2)->bool {
131  return e1.end_id() < e2.end_id();
132  });
133 
134  return paths;
135  }

137  // preparation for many to 1
138  std::deque<Path> dag(
139  G &graph,
140  const std::vector < int64_t > &start_vertex,
141  int64_t end_vertex,
142  bool only_cost) {

144 
145  for (const auto &start : start_vertex) {

152  return e1.start_id() < e2.start_id();
153  });

158  // preparation for many to many
159  std::deque<Path> dag(
160  G &graph,
161  const std::vector< int64_t > &start_vertex,
162  const std::vector< int64_t > &end_vertex,
163  bool only_cost) {
164  // a call to 1 to many is faster for each of the sources
165  std::deque<Path> paths;
166 
167  for (const auto &start : start_vertex) {
168  auto r_paths = dag(
169  graph,
170  start, end_vertex,
171  only_cost);
172  paths.insert(paths.begin(), r_paths.begin(), r_paths.end());
173  }
174 
175  std::sort(paths.begin(), paths.end(),
176  [](const Path &e1, const Path &e2)->bool {
177  return e1.end_id() < e2.end_id();
178  });
179  std::stable_sort(paths.begin(), paths.end(),
180  [](const Path &e1, const Path &e2)->bool {
181  return e1.start_id() < e2.start_id();
182  });
183  return paths;
184  }

190  bool dag_1_to_1(
191  G &graph,
192  V source,
193  V target) {
194  /* abort in case of an interruption occurs (e.g. the query is being cancelled) */
195  CHECK_FOR_INTERRUPTS();
196  try {
197  boost::dag_shortest_paths(graph.graph, source,
198  boost::predecessor_map(&predecessors[0])
199  .weight_map(get(&G::G_T_E::cost, graph.graph))
200  .distance_map(&distances[0])
201  .visitor(dijkstra_one_goal_visitor(target)));
202  } catch(found_goals &) {
203  return true;
204  } catch (boost::exception const& ex) {
205  (void)ex;
206  throw;
207  } catch (std::exception &e) {
208  (void)e;
209  throw;
210  } catch (...) {
211  throw;
212  }
213  return true;
214  }

288  class dijkstra_one_goal_visitor : public boost::default_dijkstra_visitor {
289  public:
290  explicit dijkstra_one_goal_visitor(V goal) : m_goal(goal) {}
291  template <class B_G>
292  void examine_vertex(V &u, B_G &) {
293  if (u == m_goal) throw found_goals();
294  }

300  class dijkstra_many_goal_visitor : public boost::default_dijkstra_visitor {
301  public:
302  explicit dijkstra_many_goal_visitor(
303  const std::vector< V > &goals,
304  size_t n_goals) :
305  m_goals(goals.begin(), goals.end()),
306  m_n_goals(n_goals) {}
307  template <class B_G>
308  void examine_vertex(V u, B_G &) {
309  auto s_it = m_goals.find(u);
310  if (s_it == m_goals.end()) return;
311 
312  // found one more goal
313  m_goals.erase(s_it);
314 
315  // all goals found
316  if (m_goals.size() == 0) throw found_goals();
317 
318  // number of requested goals found
319  --m_n_goals;
320  if (m_n_goals == 0) throw found_goals();
321  }
322 
323  private:
324  std::set< V > m_goals;
325  size_t m_n_goals;
326  };
