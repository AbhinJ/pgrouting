171  Path dijkstra(
172  G &graph,
173  int64_t start_vertex,
174  int64_t end_vertex,
175  bool only_cost = false) {
176  clear();
177 
178  // adjust predecessors and distances vectors
179  predecessors.resize(graph.num_vertices());
180  distances.resize(
181  graph.num_vertices(),
182  std::numeric_limits<double>::infinity());
183 
184 
185  if (!graph.has_vertex(start_vertex)
186  || !graph.has_vertex(end_vertex)) {
187  return Path(start_vertex, end_vertex);
188  }
189 
190  // get the graphs source and target
191  auto v_source(graph.get_V(start_vertex));
192  auto v_target(graph.get_V(end_vertex));
193 
194  // perform the algorithm
195  dijkstra_1_to_1(graph, v_source, v_target);
196 
197  // get the results
198  return Path(
199  graph,
200  v_source, v_target,
201  predecessors, distances,
202  only_cost, true);

211  std::deque<Path> dijkstra(
212  G &graph,
213  int64_t start_vertex,
214  const std::vector< int64_t > &end_vertex,
215  bool only_cost,
216  size_t n_goals = (std::numeric_limits<size_t>::max)()) {
217  // adjust predecessors and distances vectors
218  clear();
219 
220  predecessors.resize(graph.num_vertices());
221  distances.resize(
222  graph.num_vertices(),
223  std::numeric_limits<double>::infinity());
224 
225 
226  // get the graphs source and target
227  if (!graph.has_vertex(start_vertex))
228  return std::deque<Path>();
229  auto v_source(graph.get_V(start_vertex));
230 
231  std::set< V > s_v_targets;
232  for (const auto &vertex : end_vertex) {
233  if (graph.has_vertex(vertex)) {
234  s_v_targets.insert(graph.get_V(vertex));
235  }
236  }
237 
238  std::vector< V > v_targets(s_v_targets.begin(), s_v_targets.end());
239  // perform the algorithm
240  dijkstra_1_to_many(graph, v_source, v_targets, n_goals);
241 
242  std::deque< Path > paths;
243  // get the results // route id are the targets
244  paths = get_paths(graph, v_source, v_targets, only_cost);
245 
246  std::stable_sort(paths.begin(), paths.end(),
247  [](const Path &e1, const Path &e2)->bool {
248  return e1.end_id() < e2.end_id();
249  });
250 
251  return paths;

505  /* preparation for many to distance with equicost
506  *
507  * Idea:
508  * The distances vector does not change
509  * The predecessors vector does not change
510  * The first @b valid execution is done normally:
511  * - The distances will have:
512  * - inf
513  * - values < distance
514  * - values > distance
515  * Subsequent @b valid executions
516  * - will not change the:
517  * - values < distance
518  * Don't know yet what happens to predecessors
519  */
520  std::deque< Path > drivingDistance_with_equicost(
521  G &graph,
522  const std::vector< int64_t > &start_vertex,
523  double distance) {
524  clear();
525  log << "Number of edges:" << boost::num_edges(graph.graph) << "\n";
526 
527  predecessors.resize(graph.num_vertices());
528  distances.resize(
529  graph.num_vertices(),
530  std::numeric_limits<double>::infinity());
531 
532  /*
533  * Vector to store the different predessesors
534  * each is of size = graph.num_vertices()
535  *
536  * TODO(vicky)
537  * - figure out less storage if possible
538  */
539  std::deque< std::vector< V > > pred(start_vertex.size());
540 
541  // perform the algorithm
542  size_t i = 0;
543  for (const auto &vertex : start_vertex) {
544  nodesInDistance.clear();
545  /*
546  * The vertex does not exist
547  * Nothing to do
548  */
549  if (graph.has_vertex(vertex)) {
550  if (execute_drivingDistance_no_init(
551  graph,
552  graph.get_V(vertex),
553  distance)) {
554  pred[i] = predecessors;
555  }
556  }
557  ++i;
558  }
559 
560 
561  /*
562  * predecessors of vertices in the set are themselves
563  */
564  for (const auto &vertex : start_vertex) {
565  for (auto &p : pred) {
566  if (!p.empty() && graph.has_vertex(vertex))
567  p[graph.get_V(vertex)] = graph.get_V(vertex);
568  }
569  }
570 
571 
572  return get_drivingDistance_with_equicost_paths(
573  graph,
574  start_vertex,
575  pred,
576  distance);
577  }

586  std::deque< Path > get_drivingDistance_with_equicost_paths(
587  G &graph,
588  const std::vector< int64_t > &start_vertex,
589  std::deque< std::vector< V > > &pred,
590  double distance) {
591  /*
592  * precondition
593  */
594  pgassert(start_vertex.size() == pred.size());
595 
596 
597  /*
598  * Creating all the result "paths"
599  */
600  std::deque<Path> paths;
601  for (const auto vertex : start_vertex) {
602  paths.push_back(Path(vertex, vertex));
603  paths.back().push_back({vertex, -1, 0, 0});
604  }
605 
606  /*
607  * Ciclying the distances:
608  * To which vertex do they belong to?
609  */
610  for (V d = 0; d < distances.size(); ++d) {
611  /*
612  * Sikiping distances greater than the one asked for
613  */
614  if (!(distances[d] <= distance)) continue;
615 
616  for (auto i = start_vertex.size(); i > 0; --i) {
617  /*
618  * The vertex does not exist on the graph
619  */
620  if (pred[i - 1].empty()) break;
621 
622 
623  /*
624  * The predecessor = current then
625  * its unreachable to this vertex
626  */
627  if (pred[i - 1][d] == d) continue;
628 
629  auto cost = distances[d] - distances[pred[i - 1][d]];
630  auto edge_id = graph.get_edge_id(pred[i - 1][d], d, cost);
631  pgassert(edge_id != -1);
632  paths[i - 1].push_back(
633  {graph[d].id,
634  edge_id, cost,
635  distances[d]});
636  break;
637  }
638  }
639 
640  for (auto &path : paths) {
641  path.sort_by_node_agg_cost();
642  }
643  return paths;
644  }

746  class dijkstra_many_goal_visitor : public boost::default_dijkstra_visitor {
748  public:
749  explicit dijkstra_many_goal_visitor(
750  const std::vector< V > &goals,
751  size_t n_goals) :
752  m_goals(goals.begin(), goals.end()),
753  m_n_goals(n_goals) {}
754  template <class B_G>
755  void examine_vertex(V u, B_G &) {
756  auto s_it = m_goals.find(u);
757  if (s_it == m_goals.end()) return;
758 
759  // found one more goal
760  m_goals.erase(s_it);
761 
762  // all goals found
763  if (m_goals.size() == 0) throw found_goals();
764 
765  // number of requested goals found
766  --m_n_goals;
767  if (m_n_goals == 0) throw found_goals();
768  }
769 
770  private:
771  std::set< V > m_goals;
772  size_t m_n_goals;
773  };
