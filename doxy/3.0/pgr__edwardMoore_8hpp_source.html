47  std::deque<Path> edwardMoore(
48  G &graph,
49  std::vector<int64_t> start_vertex,
50  std::vector<int64_t> end_vertex) {
51  std::deque<Path> paths;
52 
53  for (auto source : start_vertex) {
54  std::deque<Path> result_paths = one_to_many_edwardMoore(
55  graph,
56  source,
57  end_vertex);
58 
59  paths.insert(
60  paths.begin(),
61  std::make_move_iterator(result_paths.begin()),
62  std::make_move_iterator(result_paths.end()));
63  }
64 
65  std::sort(paths.begin(), paths.end(),
66  [](const Path &e1, const Path &e2) -> bool {
67  return e1.end_id() < e2.end_id();
68  });
69  std::stable_sort(paths.begin(), paths.end(),
70  [](const Path &e1, const Path &e2) -> bool {
71  return e1.start_id() < e2.start_id();
72  });
73 
74  return paths;
75  }

80  std::deque<Path> one_to_many_edwardMoore(
81  G &graph,
82  int64_t start_vertex,
83  std::vector<int64_t> end_vertex) {
84  std::deque<Path> paths;
85 
86  if (graph.has_vertex(start_vertex) == false) {
87  return paths;
88  }
89 
90  std::vector<double> current_cost(graph.num_vertices(), std::numeric_limits<double>::infinity());
91  std::vector<bool> isInQ(graph.num_vertices(), false);
92  std::vector<E> from_edge(graph.num_vertices());
93  std::deque<int64_t> dq;
94  DEFAULT_EDGE = from_edge[0];
95 
96  int64_t bgl_start_vertex = graph.get_V(start_vertex);
97 
98  current_cost[bgl_start_vertex] = 0;
99  isInQ[bgl_start_vertex] = true;
100  dq.push_front(bgl_start_vertex);
101 
102  while (dq.empty() == false) {
103  int64_t head_vertex = dq.front();
104 
105  dq.pop_front();
106  isInQ[head_vertex] = false;
107 
108  updateVertexCosts(graph, current_cost, isInQ, from_edge, dq, head_vertex);
109  }
110 
111  for (auto target_vertex : end_vertex) {
112  if (graph.has_vertex(target_vertex) == false) {
113  continue;
114  }
115 
116  int64_t bgl_target_vertex = graph.get_V(target_vertex);
117 
118  if (from_edge[bgl_target_vertex] == DEFAULT_EDGE) {
119  continue;
120  }
121 
122  paths.push_front(
123  getPath(graph, bgl_start_vertex, target_vertex, bgl_target_vertex, from_edge, current_cost));
124  }
125 
126  return paths;
127  }

129  Path getPath(
130  G &graph,
131  int64_t bgl_start_vertex,
132  int64_t target,
133  int64_t bgl_target_vertex,
134  std::vector<E> &from_edge,
135  std::vector<double> &current_cost) {
136  int64_t current_node = bgl_target_vertex;
137 
138  Path path = Path(graph[bgl_start_vertex].id, graph[current_node].id);
139 
140  path.push_back({target, -1, 0, current_cost[current_node]});
141 
142  do {
143  E e = from_edge[current_node];
144  auto from = graph.source(e);
145 
146  path.push_back({graph[from].id, graph[e].id, graph[e].cost, current_cost[from]});
147 
148  current_node = from;
149  } while (from_edge[current_node] != DEFAULT_EDGE);
150 
151  std::reverse(path.begin(), path.end());
152  return path;
153  }

155  void updateVertexCosts(
156  G &graph,
157  std::vector<double> &current_cost,
158  std::vector<bool> &isInQ,
159  std::vector<E> &from_edge,
160  std::deque<int64_t> &dq,
161  int64_t &head_vertex) {
162  auto out_edges = boost::out_edges(head_vertex, graph.graph);
163  E e;
164  EO_i out_i;
165  EO_i out_end;
166  V v_source, v_target;
167 
168  for (boost::tie(out_i, out_end) = out_edges;
169  out_i != out_end; ++out_i) {
170  e = *out_i;
171  v_target = graph.target(e);
172  v_source = graph.source(e);
173  double edge_cost = graph[e].cost;
174 
175  if (std::isinf(current_cost[v_target]) || current_cost[v_source] + edge_cost < current_cost[v_target]) {
176  current_cost[v_target] = current_cost[v_source] + edge_cost;
177 
178  from_edge[v_target] = e;
179 
180  if (isInQ[v_target] == false) {
181  dq.push_back(v_target);
182  isInQ[v_target] = true;
183  }
184  }
185  }
186  }
