74 costCheck(G &graph) {
75  typedef typename G::E E;
76  typedef typename G::E_i E_i;
77 
78  auto edges = boost::edges(graph.graph);
79  E e;
80  E_i out_i;
81  E_i out_end;
82  std::set<double> cost_set;
83  for (boost::tie(out_i, out_end) = edges;
84  out_i != out_end; ++out_i) {
